1.建表

学生表
学号 性名 性别 出生年月日 所在班级
create table students
(
sid int primary key,
sname varchar(20) not null,
ssex varchar(2) not null,
sbirth date,
class varchar(20)
);

** 课程表 **
课程号 课程名称 教师编号
create table course
(
cid int primary key,
cname varchar(20) not null,
tid int not null,
foreign key(tid) references teachers(tid)
);

成绩表
学号 课程号 成绩
create table score
(
sid int not null,
cid int not null,
degree decimal ,
foreign key(sid) references students(sid),
foreign key(cid) references course(cid)
);

教师表
教师编号 教师名字 教师性别 出生年月日 职称 所在部门
create table teachers
(
tid int primary key,
tname varchar(20) not null,
tsex varchar(2) not null,
tbirth date,
professor varchar(20) not null,
department varchar(20) not null
);

3.查询练习

1.查询students表中所有记录
*（**:表示所有）
select *from students;

2.查询students表中的所有记录的sname， ssex，class列
select sname ,ssex,class from students;

3.查询教师所有的单位即不重复的department列 **
*（distinct:排除重复）***
select distinct(department) from teachers;

4.查询score表中成绩在50到80之间的所有记录
（between…and…:表示区间）
select * from score where degree between 50 and 80;
(运算符表示区间,and）
或者select *fron score where degree>=50 and degree <=80;

5.查询score表中成绩为60，70，80的记录 **
*（同一个字段的单独几个查询用in）***
select *from score where degree in(60,70,80);

6.查询students表中1班或性别为女的同学记录 **
*（不同字段单独关键字查询用[或：or，和：and]）***
select *from students where class=’1班’ or ssex=’女’;

7.以学号降序查询students表中的记录
（order by 字段[升序：asc；降序desc]）
select *from students order by sid asc;

8.以cid升序、degree降序查询score表中记录 **
*(这里与用，不是and）*
*这里的话，会先cid升序排序后，再在相同的cid里面对degree进行降序***
select *from score order by cno asc ,degree dasc;

9.查询1班的学生人数 **
*（统计count,放在( * * *)与from前面,记得加()）**
select count() from students where class=’1班’;

10.查询score表中最高分的学生学号和课程号
（最大值用max(字段)）
select sid,cid from score where degree=(select max(degree) from score);
*（先进行排序，再用limit进行读取）
limit第一个数字是从哪个位置开始读取(位置类似数组，第一个在位0），第二个数字是读取多少个
select sid,cid from score order by degree desc limit 0,1;

11.查询每门课的平均成绩，并把课程号打印出来（计算平均值avg，操作并显示组内成员group by）
select cid,avg(degree) from score group by cid;

12.查询score表中至少有两名学生学习的，并且课程号是2开头的课程的平均成绩和课程号
（having:达到(有)什么条件，字段 like :字段达到某一模糊条件，’3%‘表示条件是3开头后面任意，一定要单引号！）
select avg(degree),cid from score group by cid having count(cid)>=2 and cid like ‘3%’;

14.查询在成绩表中有成绩的学生的sname 、cid和degree列，也就是多表查询
（要对一个表中含有另一个表里面相同的某个字段数据的记录操作，就用 表名.字段名=表名.字段名判断并操作。✳一定要有范围约定：表名.字段名=表名.字段，不然数据没意义）
select sname,cid,degree from students,score where students.sid=score.sid;

15.查询所有学生的sname、cname和degree列，也就是三表查询
（进一步解释一下这里的where，我觉得是当一个表中的记录的该字段和另一个表中记录的该字段一样时才打印出来。✳一定要有范围约定：表名.字段名=表名.字段，不然数据没意义）
select sname,cname,degree from students,course,score where students.sid=score.sid and course.cid=score.cid;

16.查询一班学生每门课的平均分

思路
a.先找出一班的同学具有的并且记录在score中也有字段sid，因为最终还是要在score查找
select sid from students where class=’一班’;
b.然后在score表中找出这些学号的记录
select from score where score in(select sid from students where class=’一班’）；
**c.最后找自己想要的东西***
select cid,avg(degree) from score where sid in(select sid from students where class=’一班’) group by cid;

17.查询11课程的成绩高于1号同学11课程的所有同学记录
select * from score where cid=11 and degree>(select degree from score where sid=1 and cid=11);

18.查询高于学号为“2”的‘’课号为“22”的成绩的所有记录
select * from score where degree>(select degree from score where sid=2 and cid=22);

19.查询和学号为1、3的同学同年出生的所有学生的sid、sname、sbirth列
（单独拿出date类型数据的年份：year（date类型的字段））
select sid,sname,sbirth from students where year(sbirth) in (select year(sbirth) from students where sid in(1,3));

20.查询“吴祖军”教师任课学生的成绩
select *from score where cid=(select cid from course where tid=(select tid from teachers where tname=’吴祖军’));

21.查询选修某课程的同学人数不少于两人的老师名字
select tname from teachers where tid in(select tid from course where cid in(select cid from score group by cid having count(cid)>=2));

22.查询”吴祖军“教师任课的学生成绩
select degree from score where cid=(select cid from course where tid=(select tid from teachers where tname=”吴祖军”));

23.查询选修某课程的同学人数不少于两人的教师姓名
select tname from teachers where tid in(select tid from course where cid in(select cid from score group by cid having count(cid)>=2));

24.查询存在有20分以上的课程cid列
（为了不重复cid特别加了group by cid）
select cid from score where degree>=20 group by cid;

25.查询语文系教师所教课程的成绩表
（自己加了order by sid asc让它能排序美观）
select * from score where cid in(select cid from course where tid in(select tid from teachers where department=’语文系’ )) order by sid asc;

26.查询“语文系”中，“化学系”没有的职称的对应的记录 加上 “化学系”中，“语文系”没有的职称的对应的记录
（查询字段中不等于某些数值的记录 not in()，如果是查询字段中不等于某一个值就是 !=)
（合并输出：union，合并n个就加n-1个union）
select * from teachers where department=’语文系’ and professor not in(select professor from teachers where department=’化学系’)
union
select * from teachers where department=’化学系’ and professor not in(select professor from teachers where department=’语文系’);

27.查询选修编号为11课程且成绩不低于任一选修编号为22课程的同学的cid，sid，degree。并且degree以降序的方式排序
（其中任一记录：any）
select cid,sid,degree from score where cid=11 and degree > any(select degree from score where cid=22);

28.查询选修编号为11课程且成绩不低于所有选修编号为22课程的同学的cid，sid，degree。并且degree以降序的方式排序
(其中全部记录：all）
select cid,sid,degree from score where cid=11 and degree> all(select degree from score where cid=22);

29.查询所有教师和同学的name，sex，birthday列
一开始：
select tname,tsex,tbirth from teachers
union
select sname,ssex,sbirth from students;
但是
这样的话列名就是tname，tsex，birth(显示的是第一列的列名）。但是需要列名是name，sex，birth，所以
（用别名 as）
select tname as name,tsex as sex,tbirth as birth from teachers
union
select sname,ssex,sbirth from students;（下面的自动也变为name，sex，birth不用用as）
（✳union相当于并集，就是不会重复地输出数据）

30.查询成绩比该课程平均成绩低的同学记录
(复制表数据做查询条件）
把select degree from score 叫做表a
把 select avg(degree) from score group by cid 叫做表b
然后（要去掉表b的group by cid）
select * from score a where degree<(select avg(degree) from score b where a.cid=b.cid) ;

31.查询所有在分数表中有成绩的教师的tname和department
select tname,department from teachers where tid in(select tid from course where cid in(select cid from score group by cid));

32.查询至少有两名男生的班级名class
（where+条件 一定放在 having+条件之前）
select class from students where ssex=’男’ group by class having count(class)>=2;

33.查询students表中不姓古的同学记录
（字段 not ike :字段达到某一模糊条件，’古%‘表示条件是3开头后面任意，一定要单引号！）
select * from students where sname not like ‘古%’;

34.查询students每个学生的姓名和年龄
（得到当前年份：select year(now())，可以直接在查询中做运算把数值打出来）
select sname,year(now())-year(sbirth) as age from students;

35.查询students表中最大和最小的sbirth日期值
select max(sbirth) as ‘最小’ ,min(sbirth) as ‘最大’ from students;

36.以年龄从小到大的顺序查询students表中全部记录
select * from students order by sbirth asc;

37.查询男教师及其所上的课程
select cname from course where tid in(select tid from teachers where tsex=’男’);

38.查询最高分同学的sid,cid和degree
select sid,cid,degree from score where degree=(select max(degree) from score) ;

39.查询和古一同性别的除了古一之外所有同学的sname
select * from students where ssex=(select ssex from students where sname=’古一’) and sname !=’古一’;

40.查询和古一同性别和同班的除了古一之外所有同学的sname
（n个条件就加n-1个and或者or)
select * from students where ssex=(select ssex from students where sname=’古一’) and sname !=’古一’ and class=(select class from students where sname=’古一’) ;

41.查询选修大学语文课程的男同学的成绩表
select * from score where cid=(select cid from course where cname=’大学语文’) and sid in(select sid from students where ssex=’男’);

查询所有同学的sid，cid，grade列
（多表联合查询可以用其中一个表和另外的表内的列直接比较）
select sid,cid,grade from score,grade where degree between low and upp;
